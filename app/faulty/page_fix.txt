  const toggleIfOk = (id: string, currentValue: boolean) => {
    const stored = localStorage.getItem('dyeflow_db')
    if (!stored) return
    const db = JSON.parse(stored)
    const record = db.faultyRecords?.find((r: FaultyRecord) => r.id === id)
    if (record) {
      record.ifOk = !currentValue
      // If marking as OK, uncheck reprocess and set actual date to now
      if (!currentValue) {
        record.reprocess = false
        record.actualDate = new Date().toISOString()
      }
      localStorage.setItem('dyeflow_db', JSON.stringify(db))
      loadRecords()
    }
  }
